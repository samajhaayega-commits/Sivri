<script>
    const paths = {
        1: [23, 24, 19, 14, 9, 4, 3, 2, 1, 0, 5, 10, 15, 20, 21, 16, 11, 6, 7, 8, 13, 18, 17, 12],
        2: [9, 4, 3, 2, 1, 0, 5, 10, 15, 20, 21, 22, 23, 24, 19, 18, 17, 16, 11, 6, 7, 8, 13, 12],
        3: [1, 0, 5, 10, 15, 20, 21, 22, 23, 24, 19, 14, 9, 4, 3, 8, 13, 18, 17, 16, 11, 6, 7, 12],
        4: [15, 20, 21, 22, 23, 24, 19, 14, 9, 4, 3, 2, 1, 0, 5, 6, 7, 8, 13, 18, 17, 16, 11, 12]
    };

    const config = {
        1: { color: 'var(--p1)', home: 22 },
        2: { color: 'var(--p2)', home: 14 },
        3: { color: 'var(--p3)', home: 2 },
        4: { color: 'var(--p4)', home: 10 }
    };

    let turn = 1, moveVal = 0, tokens = [], winners = [], isBusy = false;

    // Audio Context for Funny sounds
    const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    function playSfx(type) {
        const osc = audioCtx.createOscillator();
        const gain = audioCtx.createGain();
        osc.connect(gain); gain.connect(audioCtx.destination);
        if(type === 'kill') { osc.type = 'sawtooth'; osc.frequency.setValueAtTime(150, audioCtx.currentTime); osc.frequency.exponentialRampToValueAtTime(40, audioCtx.currentTime + 0.4); }
        if(type === 'win') { osc.type = 'sine'; osc.frequency.setValueAtTime(500, audioCtx.currentTime); osc.frequency.exponentialRampToValueAtTime(1000, audioCtx.currentTime + 0.5); }
        gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.4);
        osc.start(); osc.stop(audioCtx.currentTime + 0.4);
    }

    function init() {
        const board = document.getElementById('board');
        const homes = { 22:1, 14:2, 2:3, 10:4 };
        for (let i = 0; i < 25; i++) {
            const cell = document.createElement('div');
            cell.className = 'cell';
            cell.id = `c${i}`;
            if (homes[i]) { cell.classList.add('home-cell'); cell.setAttribute('data-p', homes[i]); }
            if (i === 12) { cell.classList.add('win-box'); cell.innerText = "WIN"; }
            board.appendChild(cell);
        }
        for (let p = 1; p <= 4; p++) {
            for (let t = 0; t < 4; t++) {
                const el = document.createElement('div');
                el.className = 'token';
                el.style.backgroundColor = config[p].color;
                const obj = { p, t, pos: -1, el, finished: false };
                el.onclick = () => moveToken(obj);
                tokens.push(obj);
                place(obj);
            }
        }
    }

    function place(obj) {
        const cellId = obj.pos === -1 ? config[obj.p].home : paths[obj.p][obj.pos];
        const cell = document.getElementById(`c${cellId}`);
        const offset = obj.t * 3;
        obj.el.style.left = (cell.clientWidth/2 - 13 + offset) + 'px';
        obj.el.style.top = (cell.clientHeight/2 - 13 + offset) + 'px';
        cell.appendChild(obj.el);
    }

    async function toss() {
        if (isBusy) return;
        isBusy = true;
        
        // Winner Check: If all 4 are at pos 23 (Win Box entrance)
        const myTokens = tokens.filter(tk => tk.p === turn && !tk.finished);
        const allInWaiting = myTokens.every(tk => tk.pos === 23);

        const hand = document.getElementById('toss-btn');
        const seeds = document.querySelectorAll('.seed');
        hand.classList.add('shaking');
        hand.querySelector('.hand-emoji').innerText = "👐";
        
        await new Promise(r => setTimeout(r, 800));
        hand.classList.remove('shaking');
        hand.querySelector('.hand-emoji').innerText = "🖐️";

        let whites = 0;
        seeds.forEach(s => {
            const isWhite = Math.random() > 0.5;
            if (isWhite) whites++;
            const rx = isWhite ? (Math.random()*20-10) : (180 + Math.random()*20-10);
            s.style.setProperty('--rx', `${rx}deg`);
            s.classList.remove('thrown');
            void s.offsetWidth;
            s.classList.add('thrown');
        });

        moveVal = (whites === 0) ? 8 : whites;
        document.getElementById('move-info').innerText = moveVal;

        // Final Logic: Need 4 or 8 to win
        if (allInWaiting) {
            if (moveVal === 4 || moveVal === 8) {
                myTokens.forEach(tk => { tk.finished = true; tk.el.style.display = 'none'; });
                playSfx('win');
                winners.push(turn);
                showWinner(turn);
                return;
            } else {
                setTimeout(nextTurn, 1000);
                return;
            }
        }

        let canMove = false;
        tokens.forEach(tk => {
            if (tk.p === turn && !tk.finished && (tk.pos + moveVal) < paths[turn].length) {
                tk.el.classList.add('active'); canMove = true;
            }
        });
        if (!canMove) setTimeout(nextTurn, 1200);
    }

    async function moveToken(t) {
        if (!t.el.classList.contains('active')) return;
        tokens.forEach(tk => tk.el.classList.remove('active'));

        for (let i = 0; i < moveVal; i++) {
            t.pos++; place(t);
            await new Promise(r => setTimeout(r, 200));
        }

        const currentCell = paths[t.p][t.pos];
        
        // Handle Kill
        if (![22, 14, 2, 10].includes(currentCell) && currentCell !== 12) {
            let killed = false;
            tokens.forEach(other => {
                if (other.p !== t.p && !other.finished && other.pos !== -1 && paths[other.p][other.pos] === currentCell) {
                    other.pos = -1; place(other); killed = true;
                }
            });
            if (killed) { playSfx('kill'); resetBari("MARA! EXTRA TURN!"); return; }
        }

        if (currentCell === 12 || moveVal === 4 || moveVal === 8) {
            resetBari("EXTRA TURN!");
        } else {
            nextTurn();
        }
    }

    function resetBari(msg) {
        isBusy = false;
        document.getElementById('move-info').innerText = msg;
        document.getElementById('toss-btn').querySelector('.hand-emoji').innerText = "✊";
    }

    function nextTurn() {
        do { turn = (turn % 4) + 1; } while (winners.includes(turn));
        document.getElementById('status-bar').innerText = `PLAYER ${turn} TURN`;
        document.getElementById('status-bar').style.color = config[turn].color;
        isBusy = false;
        document.getElementById('move-info').innerText = "READY";
        document.getElementById('toss-btn').querySelector('.hand-emoji').innerText = "✊";
    }

    function showWinner(p) {
        document.getElementById('winner-name').innerText = `PLAYER ${p} IS SIVRI KING!`;
        document.getElementById('win-overlay').style.display = 'flex';
    }

    init();
</script>
